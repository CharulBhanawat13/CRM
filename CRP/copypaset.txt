SELECT o.ninternal_id,o.norg_id,o.corg_name,o.isAvailable,o.norg_group_id,
    Count(v.norg_id),Count(c.norg_id),Count(t.norg_id)
        FROM tbl_organisation AS o
        JOIN tbl_visitplan AS v
        ON o.norg_id=v.norg_id
        JOIN tbl_callList AS c
        ON c.norg_id=o.norg_id
        JOIN tbl_tour AS t
        ON t.norg_id=o.norg_id
       WHERE o.isAvailable=1 AND o.norg_group_id=$org_group_id
